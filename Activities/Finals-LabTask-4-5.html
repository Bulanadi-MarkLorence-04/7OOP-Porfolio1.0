Code for carClass.py File


    
class Car:
    def __init__(self, color: str, price: float, size: str):
        self.color = color
        self.price = price
        self.size = size.upper()

    def set_color(self, color: str) -> None:
        self.color = color

    def set_price(self, price: float) -> None:
        self.price = price

    def set_size(self, size: str) -> None:
        self.size = size.upper()

    def get_color(self) -> str:
        return self.color

    def get_price(self) -> float:
        if self.price >= 0:
            return self.price
        else:
            return 0

    def get_size(self) -> str:
        return self.size

    def __str__(self) -> str:
        size_desc = {"S": "small", "M": "medium", "L": "large"}.get(self.size, "unknown")
        return f"Car ({self.color}) P{self.price:.2f} {size_desc}"

def main():

    print("\nAction: Invoking the Car class constructor using Car('blue', 50000.00, 'L').")

    print("\nAction: Invoking the Car class constructor using Car('green', 12345.67, 'S').")
    car3 = Car("green", 12345.67, "S")
    print("Output:")

if __name__ == "__main__":
    main()
